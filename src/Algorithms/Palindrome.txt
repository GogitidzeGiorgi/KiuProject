num := 12321 
original := num
reversed := 0
for num != 0 {
   digit := num % 10        
   reversed = reversed*10 + digit // Append the digit to the reversed number
   num /= 10                  
}

	
if original == reversed {
   Println("%d is palindrome.", original)
} else {
   Println("%d is not palindrome.", original)
}
